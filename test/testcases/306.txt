
Class Math {
    $fact(n:Int) {
        If (n < 0) {
            Return;
        }
        Elseif (n == 0) {
            Return 1;
        }
        Else {
            Var fact: Int = 1;
            Foreach (i In 1 .. n) {
                fact = fact * i;
            }
            Return fact;
        }
    }
}

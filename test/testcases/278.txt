    $pow(a,n:Int) {
        Var powa, ans: Int = a, 1;
        Foreach(temp In 0 .. 1 By 0) {
            If (n % 2 == 1) { ans = ans * powa; }
            n = n / 2; powa = powa * powa;
            If (n == 0) { Break; }
        }
        Return ans;
    }
    main() {

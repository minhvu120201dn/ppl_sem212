
Class Shape {
    Var area: Float;
}

Class Rectangle : Shape {
    Var height, width: Float;
    Constructor(h,w:Float) {
        Self.height = h;
        Self.width = w;
        Self.area = h*w;
    }
}

Class Circle : Shape {
    Var radius: Float;
    Val $pi: Float = 3.141592653589;
    Constructor(r:Float) {
        Self.radius = r;
        Self.area = $pi*r*r;
    }
}
